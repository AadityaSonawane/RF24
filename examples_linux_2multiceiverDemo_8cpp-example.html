examples_linux/multiceiverDemo.cpp

Written by 2bndy5 in 2020

A simple example of sending data from as many as 6 nRF24L01 transceivers to 1 receiving transceiver. This technique is trademarked by Nordic Semiconductors as "MultiCeiver".

This example was written to be used on up to 6 devices acting as TX nodes & only 1 device acting as the RX node (that's a maximum of 7 devices). Use ctrl+c to quit at any time.

39 // For this example, we'll be using 6 addresses; 1 for each TX node
40 // It is very helpful to think of an address as a path instead of as
41 // an identifying device destination
42 // Notice that the last byte is the only byte that changes in the last 5
43 // addresses. This is a limitation of the nRF24L01 transceiver for pipes 2-5
44 // because they use the same first 4 bytes from pipe 1.

174 void master(unsigned int role) {
175  // set the payload's nodeID & reset the payload's identifying number
176  payload.nodeID = role;
177  payload.payloadID = 0;
178 
179  // Set the address on pipe 0 to the RX node.
180  radio.stopListening(); // put radio in TX mode
181  radio.openWritingPipe(address[role]);
182 
183  // According to the datasheet, the auto-retry features's delay value should
184  // be "skewed" to allow the RX node to receive 1 transmission at a time.
185  // So, use varying delay between retry attempts and 15 (at most) retry attempts
186  radio.setRetries(((role * 3) % 12) + 3, 15); // maximum value is 15 for both args
187 
188  unsigned int failures = 0;
189  while (failures < 6) {
190  clock_gettime(CLOCK_MONOTONIC_RAW, &startTimer); // start the timer
191  bool report = radio.write(&payload, sizeof(payload)); // transmit & save the report
192  uint32_t timerEllapsed = getMicros(); // end the timer
193 
194  if (report) {
195  // payload was delivered
196  cout << "Transmission of PayloadID ";
197  cout << payload.payloadID; // print payload number
198  cout << " as node " << payload.nodeID; // print node number
199  cout << " successful! Time to transmit = ";
200  cout << timerEllapsed << " us" << endl; // print the timer result
201  }
202  else {
203  // payload was not delivered
204  failures++;
205  cout << "Transmission failed or timed out" << endl;
206  }
207  payload.payloadID++; // increment payload number
208 
209  // to make this example readable in the terminal
210  delay(500); // slow transmissions down by 0.5 second
211  } // while
212  cout << failures << " failures detected. Leaving TX role." << endl;
213 } // master

219 void slave() {
220 
221  // Set the addresses for all pipes to TX nodes
222  for (uint8_t i = 0; i < 6; ++i)
223  radio.openReadingPipe(i, address[i]);
224 
225  radio.startListening(); // put radio in RX mode
226 
227  time_t startTimer = time(nullptr); // start a timer
228  while (time(nullptr) - startTimer < 6) { // use 6 second timeout
229  uint8_t pipe;
230  if (radio.available(&pipe)) { // is there a payload? get the pipe number that recieved it
231  uint8_t bytes = radio.getPayloadSize(); // get the size of the payload
232  radio.read(&payload, bytes); // fetch payload from FIFO
233  cout << "Received " << (unsigned int)bytes; // print the size of the payload
234  cout << " bytes on pipe " << (unsigned int)pipe; // print the pipe number
235  cout << " from node " << payload.nodeID; // print the payload's origin
236  cout << ". PayloadID: " << payload.payloadID << endl; // print the payload's number
237  startTimer = time(nullptr); // reset timer
238  }
239  }
240  cout << "Nothing received in 6 seconds. Leaving RX role." << endl;
241 } // slave

247 uint32_t getMicros() {
248  // this function assumes that the timer was started using
249  // `clock_gettime(CLOCK_MONOTONIC_RAW, &startTimer);`
250 
251  clock_gettime(CLOCK_MONOTONIC_RAW, &endTimer);
252  uint32_t seconds = endTimer.tv_sec - startTimer.tv_sec;
253  uint32_t useconds = (endTimer.tv_nsec - startTimer.tv_nsec) / 1000;
254 
255  return ((seconds) * 1000 + useconds) + 0.5;
256 }
