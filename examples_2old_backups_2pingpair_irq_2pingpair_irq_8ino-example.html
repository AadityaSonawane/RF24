examples/old_backups/pingpair_irq/pingpair_irq.ino

Updated by TMRh20

This is an example of how to user interrupts to interact with the radio, and a demonstration of how to use them to sleep when receiving, and not miss any payloads.
The pingpair_sleepy example expands on sleep functionality with a timed sleep option for the transmitter. Sleep functionality is built directly into my fork of the RF24Network library

28 // Our ACK payload will simply be 4 bytes containing the number of payloads received
29 static uint32_t message_count = 1; // start counting at 1
30 
31 // Demonstrates another method of setting up the addresses
32 byte address[][5] = {0xCC, 0xCE, 0xCC, 0xCE, 0xCC, 0xCE, 0xCC, 0xCE, 0xCC, 0xCE};

34 /************************* Role management ****************************/
35 // Set up role. This sketch uses the same software for all the nodes in this
36 // system. Doing so greatly simplifies testing.
37 
38 // The role_pin is a digital input pin used to set the role of this radio.
39 // Connect the role_pin to GND to be the 'pong' receiver
40 // Leave the role_pin open to be the 'ping' transmitter
41 const short role_pin = 5; // use pin 5
42 typedef enum { role_sender = 1, role_receiver } role_e; // The various roles supported by this sketch
43 const char* role_friendly_name[] = {"invalid", "Sender", "Receiver"}; // The debug-friendly names of those roles
44 role_e role; // The role of the current running sketch
45

47 void setup() {
48 
49  pinMode(role_pin, INPUT); // set up the role pin
50  digitalWrite(role_pin, HIGH); // Change this to LOW/HIGH instead of using an external pin
51  delay(20); // Just to get a solid reading on the role pin
52 
53  if (digitalRead(role_pin)) // read the role_pin pin to establish our role
54  role = role_sender;
55  else
56  role = role_receiver;
57 
58 
59  Serial.begin(115200);
60  printf_begin(); // needed for printDetails()
61 
62  // print introduction
63  Serial.print(F("\n\rRF24/examples/pingpair_irq\n\rROLE: "));
64  Serial.println(role_friendly_name[role]);
65 
66 
67  /********************** Setup and configure rf radio *********************/
68  radio.begin();
69 
70  // Examples are usually run with both radios in close proximity to each other
71  radio.setPALevel(RF24_PA_LOW); // defaults to RF24_PA_MAX
72  radio.enableAckPayload(); // We will be using the ACK Payload feature which is not enabled by default
73  radio.enableDynamicPayloads(); // Ack payloads are dynamic payloads
74 
75  // Open a writing and reading pipe on each radio, with opposite addresses
76  if (role == role_sender) {
77  radio.openWritingPipe(address[0]);
78  radio.openReadingPipe(1, address[1]);
79  } else {
80  radio.openWritingPipe(address[1]);
81  radio.openReadingPipe(1, address[0]);
82  radio.startListening(); // First we need to start listening
83 
84  // Add an ACK payload for the first time around; 1 is the pipe number to acknowledge
85  radio.writeAckPayload(1, &message_count, sizeof(message_count));
86  ++message_count; // increment counter by 1 for next ACK payload
87  }
88 
89  radio.printDetails(); // Dump the configuration of the rf unit for debugging
90  delay(50);
91 
92  // Attach interrupt handler to interrupt #0 (using pin 2) on BOTH the sender and receiver
93  attachInterrupt(0, check_radio, LOW);
94 } // setup

97 void loop() {
98 
99 
100  /****************** Ping Out Role ***************************/
101 
102  if (role == role_sender) {
103  // Repeatedly send the current time
104 
105  unsigned long time = millis(); // Take the time
106  Serial.print(F("Now sending "));
107  Serial.println(time);
108  radio.startWrite(&time, sizeof(unsigned long), 0); // Send the time
109  delay(2000); // Try again soon (in 2 seconds)
110  }
111 
112 
113  /****************** Pong Back Role ***************************/
114  // Receiver does nothing! All the work is in Interrupt Handler
115 
116  if (role == role_receiver) {}
117 
118 } // loop

123 void check_radio(void) {
124 
125  bool tx, fail, rx; // declare variables to store IRQ flags
126  radio.whatHappened(tx, fail, rx); // What happened?
127 
128  if (tx) { // Have we successfully transmitted?
129  if (role == role_sender)
130  Serial.println(F("Send:OK"));
131  if (role == role_receiver)
132  Serial.println(F("Ack Payload:Sent"));
133  }
134 
135  if (fail) { // Have we failed to transmit?
136  if (role == role_sender)
137  Serial.println(F("Send:Failed"));
138  if (role == role_receiver)
139  Serial.println(F("Ack Payload:Failed"));
140  }
141 
142  if (rx || radio.available()) { // Did we receive a message?
143 
144 
145 
146 
147  /**************** Ping Out Role (about received ACK payload) ************************/
148  // If we're the sender, we've received an ack payload
149  if (role == role_sender) {
150  // Get the payload and dump it
151  radio.read(&message_count, sizeof(message_count));
152  Serial.print(F("Ack: "));
153  Serial.println(message_count);
154  }
155 
156 
157  /****************** Pong Back Role ***************************/
158  // If we're the receiver, we've received a time message
159  if (role == role_receiver) {
160  // Get the payload and dump it
161 
162  static unsigned long got_time; // variable to hold the received time
163  radio.read(&got_time, sizeof(got_time)); // get the payload
164  Serial.print(F("Got payload "));
165  Serial.println(got_time);
166 
167  // Add an ACK payload for the next time around; 1 is the pipe number to acknowledge
168  radio.writeAckPayload(1, &message_count, sizeof(message_count));
169  ++message_count; // increment packet counter
170  }
171  }
172 } // check_radio
