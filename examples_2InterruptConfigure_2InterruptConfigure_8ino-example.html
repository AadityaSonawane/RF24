examples/InterruptConfigure/InterruptConfigure.ino

Written by 2bndy5 in 2020

This example uses Acknowledgement (ACK) payloads attached to ACK packets to demonstrate how the nRF24L01's IRQ (Interrupt Request) pin can be configured to detect when data is received, or when data has transmitted successfully, or when data has failed to transmit.

This example was written to be used on 2 devices acting as "nodes". Use the Serial Monitor to change each node's behavior.

55  while (!Serial) {
56  // some boards need to wait to ensure access to serial over USB
57  }

59  // initialize the transceiver on the SPI bus
60  if (!radio.begin()) {
61  Serial.println(F("radio hardware is not responding!!"));
62  while (1) {} // hold in infinite loop
63  }
64 
65  // print example's introductory prompt
66  Serial.println(F("RF24/examples/InterruptConfigure"));
67 
68  // To set the radioNumber via the Serial monitor on startup
69  Serial.println(F("Which radio is this? Enter '0' or '1'. Defaults to '0'"));
70  while (!Serial.available()) {
71  // wait for user input
72  }
73  char input = Serial.parseInt();
74  radioNumber = input == 1;
75  Serial.print(F("radioNumber = "));
76  Serial.println((int)radioNumber);
77 
78  // role variable is hardcoded to RX behavior, inform the user of this
79  Serial.println(F("*** PRESS 'T' to begin transmitting to the other node"));
80 
81  // setup the IRQ_PIN
82  pinMode(IRQ_PIN, INPUT);
83  attachInterrupt(digitalPinToInterrupt(IRQ_PIN), interruptHandler, FALLING);
84  // IMPORTANT: do not call radio.available() before calling
85  // radio.whatHappened() when the interruptHandler() is triggered by the
86  // IRQ pin FALLING event. According to the datasheet, the pipe information
87  // is unreliable during the IRQ pin FALLING transition.
88 
89  // Set the PA Level low to try preventing power supply related problems
90  // because these examples are likely run with nodes in close proximity to
91  // each other.
92  radio.setPALevel(RF24_PA_LOW); // RF24_PA_MAX is default.
93 
94  // For this example we use acknowledgment (ACK) payloads to trigger the
95  // IRQ pin when data is received on the TX node.
96  // to use ACK payloads, we need to enable dynamic payload lengths
97  radio.enableDynamicPayloads(); // ACK payloads are dynamically sized
98 
99  // Acknowledgement packets have no payloads by default. We need to enable
100  // this feature for all nodes (TX & RX) to use ACK payloads.
101  radio.enableAckPayload();
102  // Fot this example, we use the same address to send data back and forth
103 
104  // set the TX address of the RX node into the TX pipe
105  radio.openWritingPipe(address[radioNumber]); // always uses pipe 0
106 
107  // set the RX address of the TX node into a RX pipe
108  radio.openReadingPipe(1, address[!radioNumber]); // using pipe 1
109 
110  // additional setup specific to the node's role
111  if (role) {
112  // setup for TX mode
113  radio.stopListening(); // put radio in TX mode
114 
115  } else {
116  // setup for RX mode
117 
118  // let IRQ pin only trigger on "data ready" event in RX mode
119  radio.maskIRQ(1, 1, 0); // args = "data_sent", "data_fail", "data_ready"
120 
121  // Fill the TX FIFO with 3 ACK payloads for the first 3 received
122  // transmissions on pipe 1
123  radio.writeAckPayload(1, &ack_payloads[0], ack_pl_size);
124  radio.writeAckPayload(1, &ack_payloads[1], ack_pl_size);
125  radio.writeAckPayload(1, &ack_payloads[2], ack_pl_size);
126 
127  radio.startListening(); // put radio in RX mode
128  }

137 void loop() {
138  if (role && !wait_for_event) {
139 
140  // delay(1); // wait for IRQ pin to fully RISE
141 
142  // This device is a TX node. This if block is only triggered when
143  // NOT waiting for an IRQ event to happen
144 
145  if (pl_iterator == 0) {
146  // Test the "data ready" event with the IRQ pin
147 
148  Serial.println(F("\nConfiguring IRQ pin to ignore the 'data sent' event"));
149  radio.maskIRQ(true, false, false); // args = "data_sent", "data_fail", "data_ready"
150  Serial.println(F(" Pinging RX node for 'data ready' event..."));
151 
152  } else if (pl_iterator == 1) {
153  // Test the "data sent" event with the IRQ pin
154 
155  Serial.println(F("\nConfiguring IRQ pin to ignore the 'data ready' event"));
156  radio.maskIRQ(false, false, true); // args = "data_sent", "data_fail", "data_ready"
157  Serial.println(F(" Pinging RX node for 'data sent' event..."));
158 
159  } else if (pl_iterator == 2) {
160  // Use this iteration to fill the RX node's FIFO which sets us up for the next test.
161 
162  // write() uses virtual interrupt flags that work despite the masking of the IRQ pin
163  radio.maskIRQ(1, 1, 1); // disable IRQ masking for this step
164 
165  Serial.println(F("\nSending 1 payload to fill RX node's FIFO. IRQ pin is neglected."));
166  // write() will call flush_tx() on 'data fail' events
167  if (radio.write(&tx_payloads[pl_iterator], tx_pl_size)) {
168  if (radio.rxFifoFull()) {
169  Serial.println(F("RX node's FIFO is full; it is not listening any more"));
170  } else {
171  Serial.println("Transmission successful, but the RX node might still be listening.");
172  }
173  } else {
174  Serial.println(F("Transmission failed or timed out. Continuing anyway."));
175  radio.flush_tx(); // discard payload(s) that failed to transmit
176  }
177 
178  } else if (pl_iterator == 3) {
179  // test the "data fail" event with the IRQ pin
180 
181  Serial.println(F("\nConfiguring IRQ pin to reflect all events"));
182  radio.maskIRQ(0, 0, 0); // args = "data_sent", "data_fail", "data_ready"
183  Serial.println(F(" Pinging inactive RX node for 'data fail' event..."));
184  }
185 
186  if (pl_iterator < 4 && pl_iterator != 2) {
187 
188  // IRQ pin is LOW when activated. Otherwise it is always HIGH
189  // Wait until IRQ pin is activated.
190  wait_for_event = true;
191 
192  // use the non-blocking call to write a payload and begin transmission
193  // the "false" argument means we are expecting an ACK packet response
194  radio.startFastWrite(tx_payloads[pl_iterator++], tx_pl_size, false);
195 
196  // In this example, the "data fail" event is always configured to
197  // trigger the IRQ pin active. Because the auto-ACK feature is on by
198  // default, we don't need a timeout check to prevent an infinite loop.
199 
200  } else if (pl_iterator == 4) {
201  // all IRQ tests are done; flush_tx() and print the ACK payloads for fun
202 
203  // CE pin is still HIGH which consumes more power. Example is now idling so...
204  radio.stopListening(); // ensure CE pin is LOW
205  // stopListening() also calls flush_tx() when ACK payloads are enabled
206 
207  printRxFifo();
208  pl_iterator++;
209 
210 
211  // inform user what to do next
212  Serial.println(F("\n*** PRESS 'T' to restart the transmissions"));
213  Serial.println(F("*** PRESS 'R' to change to Receive role\n"));
214 
215 
216  } else if (pl_iterator == 2) {
217  pl_iterator++; // proceed from step 3 to last step (stop at step 4 for readability)
218  }
219 
220  } else if (!role) {
221  // This device is a RX node
222 
223  if (radio.rxFifoFull()) {
224  // wait until RX FIFO is full then stop listening
225 
226  delay(100); // let ACK payload finish transmitting
227  radio.stopListening(); // also discards unused ACK payloads
228  printRxFifo(); // flush the RX FIFO
229 
230  // Fill the TX FIFO with 3 ACK payloads for the first 3 received
231  // transmissions on pipe 1.
232  radio.writeAckPayload(1, &ack_payloads[0], ack_pl_size);
233  radio.writeAckPayload(1, &ack_payloads[1], ack_pl_size);
234  radio.writeAckPayload(1, &ack_payloads[2], ack_pl_size);
235 
236  delay(100); // let TX node finish its role
237  radio.startListening(); // We're ready to start over. Begin listening.
238  }
239 
240  } // role
241 
242  if (Serial.available()) {
243  // change the role via the serial monitor
244 
245  char c = toupper(Serial.read());
246  if (c == 'T') {
247  // Become the TX node
248  if (!role)
249  Serial.println(F("*** CHANGING TO TRANSMIT ROLE -- PRESS 'R' TO SWITCH BACK"));
250  else
251  Serial.println(F("*** RESTARTING IRQ PIN TEST ***"));
252 
253  role = true;
254  wait_for_event = false;
255  pl_iterator = 0; // reset the iterator
256  radio.flush_tx(); // discard any payloads in the TX FIFO
257 
258  // startListening() clears the IRQ masks also. This is required for
259  // continued TX operations when a transmission fails.
260  radio.stopListening(); // this also discards any unused ACK payloads
261 
262  } else if (c == 'R' && role) {
263  // Become the RX node
264  Serial.println(F("*** CHANGING TO RECEIVE ROLE -- PRESS 'T' TO SWITCH BACK"));
265 
266  role = false;
267 
268  radio.maskIRQ(0, 0, 0); // the IRQ pin should only trigger on "data ready" event
269 
270  // Fill the TX FIFO with 3 ACK payloads for the first 3 received
271  // transmissions on pipe 1
272  radio.flush_tx(); // make sure there is room for 3 new ACK payloads
273  radio.writeAckPayload(1, &ack_payloads[0], ack_pl_size);
274  radio.writeAckPayload(1, &ack_payloads[1], ack_pl_size);
275  radio.writeAckPayload(1, &ack_payloads[2], ack_pl_size);
276  radio.startListening();
277  }
278  } // Serial.available()
279 } // loop

285 void interruptHandler() {
286  // print IRQ status and all masking flags' states
287 
288  Serial.println(F("\tIRQ pin is actively LOW")); // show that this function was called
289  delayMicroseconds(250);
290  bool tx_ds, tx_df, rx_dr; // declare variables for IRQ masks
291  radio.whatHappened(tx_ds, tx_df, rx_dr); // get values for IRQ masks
292  // whatHappened() clears the IRQ masks also. This is required for
293  // continued TX operations when a transmission fails.
294  // clearing the IRQ masks resets the IRQ pin to its inactive state (HIGH)
295 
296  Serial.print(F("\tdata_sent: "));
297  Serial.print(tx_ds); // print "data sent" mask state
298  Serial.print(F(", data_fail: "));
299  Serial.print(tx_df); // print "data fail" mask state
300  Serial.print(F(", data_ready: "));
301  Serial.println(rx_dr); // print "data ready" mask state
302 
303  if (tx_df) // if TX payload failed
304  radio.flush_tx(); // clear all payloads from the TX FIFO
305 
306  // print if test passed or failed. Unintentional fails mean the RX node was not listening.
307  // pl_iterator has already been incremented by now
308  if (pl_iterator <= 1) {
309  Serial.print(F(" 'Data Ready' event test "));
310  Serial.println(rx_dr ? F("passed") : F("failed"));
311  } else if (pl_iterator == 2) {
312  Serial.print(F(" 'Data Sent' event test "));
313  Serial.println(tx_ds ? F("passed") : F("failed"));
314  } else if (pl_iterator == 4) {
315  Serial.print(F(" 'Data Fail' event test "));
316  Serial.println(tx_df ? F("passed") : F("failed"));
317  }
318  wait_for_event = false; // ready to continue with loop() operations
319 } // interruptHandler

326 void printRxFifo() {
327  if (radio.available()) { // if there is data in the RX FIFO
328  // to flush the data from the RX FIFO, we'll fetch it all using 1 buffer
329 
330  uint8_t pl_size = !role ? tx_pl_size : ack_pl_size;
331  char rx_fifo[pl_size * 3 + 1]; // RX FIFO is full & we know ACK payloads' size
332  if (radio.rxFifoFull()) {
333  rx_fifo[pl_size * 3] = 0; // add a NULL terminating char to use as a c-string
334  radio.read(&rx_fifo, pl_size * 3); // this clears the RX FIFO (for this example)
335  } else {
336  uint8_t i = 0;
337  while (radio.available()) {
338  radio.read(&rx_fifo + (i * pl_size), pl_size);
339  i++;
340  }
341  rx_fifo[i * pl_size] = 0; // add a NULL terminating char to use as a c-string
342  }
343  Serial.print(F("Complete RX FIFO: "));
344  Serial.println(rx_fifo); // print the entire RX FIFO with 1 buffer
345  }
