examples_linux/manualAcknowledgements.cpp

Written by 2bndy5 in 2020

A simple example of sending data from 1 nRF24L01 transceiver to another with manually transmitted (non-automatic) Acknowledgement (ACK) payloads. This example still uses ACK packets, but they have no payloads. Instead the acknowledging response is sent with write(). This tactic allows for more updated acknowledgement payload data, where actual ACK payloads' data are outdated by 1 transmission because they have to loaded before receiving a transmission.

This example was written to be used on 2 devices acting as "nodes". Use ctrl+c to quit at any time.

143void master()
144{
145
146 memcpy(payload.message, "Hello ", 6); // set the outgoing message
147 radio.stopListening(); // put in TX mode
148
149 unsigned int failures = 0; // keep track of failures
150 while (failures < 6) {
151 clock_gettime(CLOCK_MONOTONIC_RAW, &startTimer); // start the timer
152 bool report = radio.write(&payload, sizeof(payload)); // transmit & save the report
153
154 if (report) {
155 // transmission successful; wait for response and print results
156
157 radio.startListening(); // put in RX mode
158 unsigned long start_timeout = millis(); // timer to detect no response
159 while (!radio.available()) { // wait for response
160 if (millis() - start_timeout > 200) // only wait 200 ms
161 break;
162 }
163 unsigned long ellapsedTime = getMicros(); // end the timer
164 radio.stopListening(); // put back in TX mode
165
166 // print summary of transactions
167 uint8_t pipe;
168 cout << "Transmission successful! ";
169 if (radio.available(&pipe)) { // is there a payload received? grab the pipe number that received it
170 uint8_t bytes = radio.getPayloadSize(); // grab the incoming payload size
171 cout << "Round trip delay = ";
172 cout << ellapsedTime; // print the timer result
173 cout << " us. Sent: " << payload.message; // print outgoing message
174 cout << (unsigned int)payload.counter; // print outgoing counter
175 PayloadStruct received;
176 radio.read(&received, sizeof(received)); // get incoming payload
177 cout << " Recieved " << (unsigned int)bytes; // print incoming payload size
178 cout << " on pipe " << (unsigned int)pipe; // print RX pipe number
179 cout << ": " << received.message; // print the incoming message
180 cout << (unsigned int)received.counter; // print the incoming counter
181 cout << endl;
182 payload.counter = received.counter; // save incoming counter for next outgoing counter
183 }
184 else {
185 cout << "Recieved no response." << endl; // no response received
186 }
187 }
188 else {
189 cout << "Transmission failed or timed out"; // payload was not delivered
190 cout << endl;
191 failures++; // increment failure counter
192 } // report
193
194 // to make this example readable in the terminal
195 delay(1000); // slow transmissions down by 1 second
196 } // while

198 cout << failures << " failures detected. Leaving TX role." << endl;

204void slave()
205{
206 memcpy(payload.message, "World ", 6); // set the response message
207 radio.startListening(); // put in RX mode
208
209 time_t startTimer = time(nullptr); // start a timer
210 while (time(nullptr) - startTimer < 6) { // use 6 second timeout
211 uint8_t pipe;
212 if (radio.available(&pipe)) { // is there a payload? get the pipe number that recieved it
213 uint8_t bytes = radio.getPayloadSize(); // get size of incoming payload
214 PayloadStruct received;
215 radio.read(&received, sizeof(received)); // get incoming payload
216 payload.counter = received.counter + 1; // increment payload for response
217
218 // transmit response & save result to `report`
219 radio.stopListening(); // put in TX mode
220 radio.writeFast(&payload, sizeof(payload)); // load response into TX FIFO
221 bool report = radio.txStandBy(150); // keep retrying for 150 ms
222 radio.startListening(); // put back in RX mode
223
224 // print summary of transactions
225 cout << "Received " << (unsigned int)bytes; // print the size of the payload
226 cout << " bytes on pipe ";
227 cout << (unsigned int)pipe; // print the pipe number
228 cout << ": " << received.message; // print incoming message
229 cout << (unsigned int)received.counter; // print incoming counter
230
231 if (report) {
232 cout << " Sent: " << payload.message; // print outgoing message
233 cout << (unsigned int)payload.counter; // print outgoing counter
234 cout << endl;
235 }
236 else {
237 cout << " Response failed to send." << endl; // failed to send response
238 }
239 startTimer = time(nullptr); // reset timer
240 } // available
241 } // while
242
243 cout << "Nothing received in 6 seconds. Leaving RX role." << endl;
244 radio.stopListening(); // recommended idle mode is TX mode
245} // slave

250uint32_t getMicros()
251{
252 // this function assumes that the timer was started using
253 // `clock_gettime(CLOCK_MONOTONIC_RAW, &startTimer);`
254
255 clock_gettime(CLOCK_MONOTONIC_RAW, &endTimer);
256 uint32_t seconds = endTimer.tv_sec - startTimer.tv_sec;
257 uint32_t useconds = (endTimer.tv_nsec - startTimer.tv_nsec) / 1000;
258
259 return ((seconds)*1000 + useconds) + 0.5;
260}
