examples_linux/streamingData.cpp

Written by 2bndy5 in 2020

A simple example of sending data from 1 nRF24L01 transceiver to another.

This example was written to be used on 2 devices acting as "nodes". Use ctrl+c to quit at any time.

52int main(int argc, char** argv)
53{
54
55 // perform hardware check
56 if (!radio.begin()) {
57 cout << "radio hardware is not responding!!" << endl;
58 return 0; // quit now
59 }
60
61 // add a NULL terminating 0 for printing as a c-string
62 buffer[SIZE] = 0;
63
64 // Let these addresses be used for the pair of nodes used in this example
65 uint8_t address[2][6] = {"1Node", "2Node"};
66 // the TX address^ , ^the RX address
67 // It is very helpful to think of an address as a path instead of as
68 // an identifying device destination
69
70 // to use different addresses on a pair of radios, we need a variable to
71 // uniquely identify which address this radio will use to transmit
72 bool radioNumber = 1; // 0 uses address[0] to transmit, 1 uses address[1] to transmit
73
74 bool foundArgNode = false;
75 bool foundArgRole = false;
76 bool role = false;
77 if (argc > 1) {
78 // CLI args are specified
79 if ((argc - 1) % 2 != 0) {
80 // some CLI arg doesn't have an option specified for it
81 printHelp(string(argv[0])); // all args need an option in this example
82 return 0;
83 }
84 else {
85 // iterate through args starting after program name
86 int a = 1;
87 while (a < argc) {
88 bool invalidOption = false;
89 if (strcmp(argv[a], "-n") == 0 || strcmp(argv[a], "--node") == 0) {
90 // "-n" or "--node" has been specified
91 foundArgNode = true;
92 if (argv[a + 1][0] - 48 <= 1) {
93 radioNumber = (argv[a + 1][0] - 48) == 1;
94 }
95 else {
96 // option is invalid
97 invalidOption = true;
98 }
99 }
100 else if (strcmp(argv[a], "-r") == 0 || strcmp(argv[a], "--role") == 0) {
101 // "-r" or "--role" has been specified
102 foundArgRole = true;
103 if (argv[a + 1][0] - 48 <= 1) {
104 role = (argv[a + 1][0] - 48) == 1;
105 }
106 else {
107 // option is invalid
108 invalidOption = true;
109 }
110 }
111 if (invalidOption) {
112 printHelp(string(argv[0]));
113 return 0;
114 }
115 a += 2;
116 } // while
117 if (!foundArgNode && !foundArgRole) {
118 // no valid args were specified
119 printHelp(string(argv[0]));
120 return 0;
121 }
122 } // else
123 } // if
124
125 // print example's name
126 cout << argv[0] << endl;
127
128 if (!foundArgNode) {
129 // Set the radioNumber via the terminal on startup
130 cout << "Which radio is this? Enter '0' or '1'. Defaults to '0' ";
131 string input;
132 getline(cin, input);
133 radioNumber = input.length() > 0 && (uint8_t)input[0] == 49;
134 }
135
136 // save on transmission time by setting the radio to only transmit the
137 // number of bytes we need to transmit a float
138 radio.setPayloadSize(SIZE); // default value is the maximum 32 bytes
139
140 // Set the PA Level low to try preventing power supply related problems
141 // because these examples are likely run with nodes in close proximity to
142 // each other.
143 radio.setPALevel(RF24_PA_LOW); // RF24_PA_MAX is default.
144
145 // set the TX address of the RX node into the TX pipe
146 radio.openWritingPipe(address[radioNumber]); // always uses pipe 0
147
148 // set the RX address of the TX node into a RX pipe
149 radio.openReadingPipe(1, address[!radioNumber]); // using pipe 1
150
151 // For debugging info
152 // radio.printDetails(); // (smaller) function that prints raw register values
153 // radio.printPrettyDetails(); // (larger) function that prints human readable data
154
155 // ready to execute program now
156 if (!foundArgRole) { // if CLI arg "-r"/"--role" was not specified
157 setRole(); // calls master() or slave() based on user input
158 }
159 else { // if CLI arg "-r"/"--role" was specified
160 role ? master() : slave(); // based on CLI arg option
161 }
162 return 0;

169void setRole()
170{
171 string input = "";
172 while (!input.length()) {
173 cout << "*** PRESS 'T' to begin transmitting to the other node\n";
174 cout << "*** PRESS 'R' to begin receiving from the other node\n";
175 cout << "*** PRESS 'Q' to exit" << endl;
176 getline(cin, input);
177 if (input.length() >= 1) {
178 if (input[0] == 'T' || input[0] == 't')
179 master();
180 else if (input[0] == 'R' || input[0] == 'r')
181 slave();
182 else if (input[0] == 'Q' || input[0] == 'q')
183 break;
184 else
185 cout << input[0] << " is an invalid input. Please try again." << endl;
186 }
187 input = ""; // stay in the while loop
188 } // while
189} // setRole()

194void master()
195{
196 radio.stopListening(); // put radio in TX mode
197
198 unsigned int failures = 0; // keep track of failures
199 uint8_t i = 0;
200 clock_gettime(CLOCK_MONOTONIC_RAW, &startTimer); // start the timer
201 while (i < SIZE) {
202 makePayload(i);
203 if (!radio.writeFast(&buffer, SIZE)) {
204 failures++;
205 radio.reUseTX();
206 }
207 else {
208 i++;
209 }
210
211 if (failures >= 100) {
212 // most likely no device is listening for the data stream
213 cout << "Too many failures detected. ";
214 cout << "Aborting at payload " << buffer[0];
215 break;
216 }
217 } // while
218 uint32_t ellapsedTime = getMicros(); // end the timer
219 cout << "Time to transmit data = ";
220 cout << ellapsedTime; // print the timer result
221 cout << " us. " << failures; // print number of retries
222 cout << " failures detected. Leaving TX role." << endl;
223} // master

228void slave()
229{
230
231 counter = 0;
232 radio.startListening(); // put radio in RX mode
233 time_t startTimer = time(nullptr); // start a timer
234 while (time(nullptr) - startTimer < 6) { // use 6 second timeout
235 if (radio.available()) { // is there a payload
236 radio.read(&buffer, SIZE); // fetch payload from FIFO
237 cout << "Received: " << buffer; // print the payload's value
238 cout << " - " << counter << endl; // print the counter
239 counter++; // increment counter
240 startTimer = time(nullptr); // reset timer
241 }
242 }
243 radio.stopListening(); // use TX mode for idle behavior
244
245 cout << "Nothing received in 6 seconds. Leaving RX role." << endl;
246}

252void makePayload(uint8_t i)
253{
254
255 // let the first character be an identifying alphanumeric prefix
256 // this lets us see which payload didn't get received
257 buffer[0] = i + (i < 26 ? 65 : 71);
258 for (uint8_t j = 0; j < SIZE - 1; ++j) {
259 char chr = j >= (SIZE - 1) / 2 + abs((SIZE - 1) / 2 - i);
260 chr |= j < (SIZE - 1) / 2 - abs((SIZE - 1) / 2 - i);
261 buffer[j + 1] = chr + 48;
262 }
263}

268uint32_t getMicros()
269{
270 // this function assumes that the timer was started using
271 // `clock_gettime(CLOCK_MONOTONIC_RAW, &startTimer);`
272
273 clock_gettime(CLOCK_MONOTONIC_RAW, &endTimer);
274 uint32_t seconds = endTimer.tv_sec - startTimer.tv_sec;
275 uint32_t useconds = (endTimer.tv_nsec - startTimer.tv_nsec) / 1000;
276
277 return ((seconds)*1000 + useconds) + 0.5;
278}
