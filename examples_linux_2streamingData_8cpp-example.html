examples_linux/streamingData.cpp

Written by 2bndy5 in 2020

A simple example of sending data from 1 nRF24L01 transceiver to another.

This example was written to be used on 2 devices acting as "nodes". Use ctrl+c to quit at any time.

195 void master() {
196  radio.stopListening(); // put radio in TX mode
197 
198  unsigned int failures = 0; // keep track of failures
199  uint8_t i = 0;
200  clock_gettime(CLOCK_MONOTONIC_RAW, &startTimer); // start the timer
201  while (i < SIZE) {
202  makePayload(i);
203  if (!radio.writeFast(&buffer, SIZE)) {
204  failures++;
205  radio.reUseTX();
206  } else {
207  i++;
208  }
209 
210  if (failures >= 100) {
211  // most likely no device is listening for the data stream
212  cout << "Too many failures detected. ";
213  cout << "Aborting at payload " << buffer[0];
214  break;
215  }
216  } // while
217  uint32_t ellapsedTime = getMicros(); // end the timer
218  cout << "Time to transmit data = ";
219  cout << ellapsedTime; // print the timer result
220  cout << " us. " << failures; // print number of retries
221  cout << " failures detected. Leaving TX role." << endl;
222 } // master

228 void slave() {
229 
230  counter = 0;
231  radio.startListening(); // put radio in RX mode
232  time_t startTimer = time(nullptr); // start a timer
233  while (time(nullptr) - startTimer < 6) { // use 6 second timeout
234  if (radio.available()) { // is there a payload
235  radio.read(&buffer, SIZE); // fetch payload from FIFO
236  cout << "Received: " << buffer; // print the payload's value
237  cout << " - " << counter << endl; // print the counter
238  counter++; // increment counter
239  startTimer = time(nullptr); // reset timer
240  }
241  }
242  radio.stopListening(); // use TX mode for idle behavior
243 
244  cout << "Nothing received in 6 seconds. Leaving RX role." << endl;
245 }

252 void makePayload(uint8_t i) {
253 
254  // let the first character be an identifying alphanumeric prefix
255  // this lets us see which payload didn't get received
256  buffer[0] = i + (i < 26 ? 65 : 71);
257  for (uint8_t j = 0; j < SIZE - 1; ++j) {
258  char chr = j >= (SIZE - 1) / 2 + abs((SIZE - 1) / 2 - i);
259  chr |= j < (SIZE - 1) / 2 - abs((SIZE - 1) / 2 - i);
260  buffer[j + 1] = chr + 48;
261  }
262 }

268 uint32_t getMicros() {
269  // this function assumes that the timer was started using
270  // `clock_gettime(CLOCK_MONOTONIC_RAW, &startTimer);`
271 
272  clock_gettime(CLOCK_MONOTONIC_RAW, &endTimer);
273  uint32_t seconds = endTimer.tv_sec - startTimer.tv_sec;
274  uint32_t useconds = (endTimer.tv_nsec - startTimer.tv_nsec) / 1000;
275 
276  return ((seconds) * 1000 + useconds) + 0.5;
277 }
