examples_linux/manual_acknowledgements.py

Written by 2bndy5 in 2020

This is a simple example of using the RF24 class on a Raspberry Pi to transmit and respond with acknowledgment (ACK) transmissions. Notice that the auto-ack feature is enabled, but this example doesn't use automatic ACK payloads because automatic ACK payloads' data will always be outdated by 1 transmission. Instead, this example uses a call and response paradigm.

Remember to install the Python wrapper, then navigate to the "RF24/examples_linux" folder.
To run this example, enter

python3 manual_acknowledgements.py

and follow the prompts.

Note
this example requires python v3.7 or newer because it measures transmission time with time.monotonic_ns().

59 def master():
60  """Transmits a message and an incrementing integer every second, then
61  wait for a response for up to 200 ms.
62  """
63  radio.stopListening() # put radio in TX mode
64  failures = 0
65  while failures < 6:
66  # use bytes() to pack our counter data into the payload
67  # NOTE b"\x00" byte is a c-string's NULL terminating 0
68  buffer = b"Hello \x00" + bytes(counter)
69  start_timer = time.monotonic_ns() # start timer
70  result = radio.write(buffer)
71  if not result:
72  failures += 1
73  print("Transmission failed or timed out")
74  else:
75  radio.startListening() # put radio in RX mode
76  timout = time.monotonic() * 1000 + 200 # use 200 ms timeout
77  # declare a variable to save the incoming response
78  while not radio.available() and time.monotonic() * 1000 < timout:
79  pass # wait for incoming payload or timeout
80  radio.stopListening() # put radio in TX mode
81  end_timer = time.monotonic_ns() # end timer
82  print(
83  "Transmission successful. Sent: {}{}.".format(
84  buffer[:6].decode("utf-8"),
85  counter[0]
86  ),
87  end=" "
88  )
89  has_payload, pipe_number = radio.available_pipe()
90  if has_payload:
91  # grab the incoming payload
92  received = radio.read(radio.payloadSize)
93  # NOTE received[7:8] discards NULL terminating 0
94  counter[0] = received[7:8][0] # save the counter
95  print(
96  "Received {} bytes on pipe {}: {}{}. "
97  "Round-trip delay: {} us.".format(
98  radio.payloadSize,

103  )
104  )

111 def slave(timeout=6):
112  """Listen for any payloads and print the transaction
113 
114  :param int timeout: The number of seconds to wait (with no transmission)
115  until exiting function.
116  """
117  radio.startListening() # put radio in RX mode
118 
119  start_timer = time.monotonic() # start a timer to detect timeout
120  while (time.monotonic() - start_timer) < timeout:
121  # receive `count` payloads or wait 6 seconds till timing out
122  has_payload, pipe_number = radio.available_pipe()
123  if has_payload:
124  received = radio.read(radio.payloadSize) # fetch the payload
125  # NOTE received[7:8] discards NULL terminating 0
126  # increment the counter from received payload
127  counter[0] = received[7:8][0] + 1 if received[7:8][0] < 255 else 0
128  # use bytes() to pack our counter data into the payload
129  # NOTE b"\x00" byte is a c-string's NULL terminating 0
130  buffer = b"World \x00" + bytes(counter)
131  radio.stopListening() # put radio in TX mode
132  radio.writeFast(buffer) # load response into TX FIFO
133  # keep retrying to send response for 150 milliseconds
134  result = radio.txStandBy(150) # save response's result
135  # NOTE txStandBy() flushes TX FIFO on transmission failure
136  radio.startListening() # put radio back in RX mode
137  # print the payload received payload
138  print(
139  "Received {} bytes on pipe {}: {}{}.".format(
140  radio.payloadSize,
141  pipe_number,
142  bytes(received[:6]).decode("utf-8"),
143  received[7:8][0]
144  ),
145  end=" "
146  )
147  if result: # did response succeed?
148  # print response's payload
149  print(
150  "Sent: {}{}".format(
151  buffer[:6].decode("utf-8"),
152  counter[0]
153  )
154  )
155  else:
